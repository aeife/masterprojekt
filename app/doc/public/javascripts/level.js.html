¶ initializeGrid

Erzeugung des Spielfeldes

Params
gridSize int Größe des zu erzeugenden Spielfeldes 
  initializeGrid: function(gridSize){
    this.gridSize = gridSize;

    for (var i = 0; i < this.gridSize; i++) {
        this.grid[i] = [];
        for (var j = 0; j < this.gridSize; j++){
            this.grid[i][j] = {food: 0, player: false};
        }
    }
    
    this.printGrid();
  },

¶ isInGrid

Params
x int x-Koordinate der zu prüfenden Kachel
y int y-Koordinate der zu prüfenden Kachel 
  isInGrid: function(x, y){
    if (x < 0 || x >= this.gridSize || y < 0 || y >= this.gridSize){
          return false;
      } else{
          return true;
      }
  },
