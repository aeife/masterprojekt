    searchPlayerById
  




  

  
Suche eines bestimmten Spielers im Spielerfeld anhand seiner ID

  

  

    
Params

    

      id
      int
      ID des gesuchten Spielers
    

  



        		
        
function searchPlayerById(id){
    var playerPos = -1;
    for (var i = 0; i < players.length; i++){
        if (players[i].id == id){
            playerPos = i;
        }
    }

    return players[playerPos];
}
